from macq import generate, extract
from macq.observation import IdentityObservation

traces = generate.pddl.VanillaSampling(problem_id=123, plan_len=20, num_traces=100).traces
observations = traces.tokenize(IdentityObservation)
model = extract.Extract(observations, extract.modes.OBSERVER)

print(model.details())

ARMS

Usage

from macq import generate, extract
from macq.trace import PlanningObject, Fluent, TraceList
from macq.observation import PartialObservation

def get_fluent(name: str, objs: list[str]):
    objects = [PlanningObject(o.split()[0], o.split()[1]) for o in objs]
    return Fluent(name, objects)

traces = TraceList()
generator = generate.pddl.TraceFromGoal(problem_id=1801)

generator.change_goal(
    {
        get_fluent("communicated_soil_data", ["waypoint waypoint2"]),
        get_fluent("communicated_rock_data", ["waypoint waypoint3"]),
        get_fluent(
            "communicated_image_data", ["objective objective1", "mode high_res"]
        ),
    }
)
traces.append(generator.generate_trace())

generator.change_goal(
    {
        get_fluent("communicated_soil_data", ["waypoint waypoint2"]),
        get_fluent("communicated_rock_data", ["waypoint waypoint3"]),
        get_fluent(
            "communicated_image_data", ["objective objective1", "mode high_res"]
        ),
    }
)
traces.append(generator.generate_trace())

observations = traces.tokenize(PartialObservation, percent_missing=0.60)
model = extract.Extract(
    observations,
    extract.modes.ARMS,
    upper_bound=2,
    min_support=2,
    action_weight=110,
    info_weight=100,
    threshold=0.6,
    info3_default=30,
    plan_default=30,
)

print(model.details())

class Model: 
 18class Model:
 19    """Action model.
 20
 21    An action model representing a planning domain. The characteristics of the
 22    model are dependent on the extraction technique used to obtain the model.
 23
 24    Attributes:
 25        fluents (set):
 26            The set of fluents in the domain.
 27        actions (set):
 28            The set of actions in the domain. Actions include their
 29            preconditions, add effects, and delete effects. The nature of the
 30            action attributes characterize the model.
 31    """
 32
 33    def __init__(
 34        self,
 35        fluents: Union[Set[LearnedFluent], Set[LearnedLiftedFluent]],
 36        actions: Union[Set[LearnedAction], Set[LearnedLiftedAction]],
 37    ):
 38        """Initializes a Model with a set of fluents and a set of actions.
 39
 40        Args:
 41            fluents (Union[Set[LearnedFluent], Set[str]]):
 42                The set of fluents in the model.
 43            actions (Set[LearnedAction]):
 44                The set of actions in the model.
 45        """
 46        self.fluents = fluents
 47        self.actions = actions
 48
 49    def __eq__(self, other):
 50        if not isinstance(other, Model):
 51            return False
 52        self_fluent_type, other_fluent_type = type(list(self.fluents)[0]), type(
 53            list(other.fluents)[0]
 54        )
 55        if self_fluent_type == other_fluent_type:
 56            return self.fluents == other.fluents and self.actions == other.actions
 57        if self_fluent_type == str:
 58            return set(map(lambda f: str(f), other.fluents)) == self.fluents
 59        if other_fluent_type == str:
 60            return set(map(lambda f: str(f), self.fluents)) == other.fluents
 61
 62    def details(self):
 63        # Set the indent width
 64        indent = " " * 2
 65        string = "Model:\n"
 66        # Map fluents to a comma separated string of the fluent names
 67        string += f"{indent}Fluents: {', '.join(map(str,self.fluents))}\n"
 68
 69        # Map the actions to a summary of their names, preconditions, add
 70        # effects and delete effects
 71        string += f"{indent}Actions:\n"
 72        for line in self._get_action_details().splitlines():
 73            string += f"{indent * 2}{line}\n"
 74        return string
 75
 76    def _get_action_details(self):
 77        # Set the indent width
 78        indent = " " * 2
 79        details = ""
 80        for action in self.actions:
 81            details += action.details() + ":\n"
 82            details += f"{indent}precond:\n"
 83            for f in action.precond:
 84                details += f"{indent * 2}{f}\n"
 85            details += f"{indent}add:\n"
 86            for f in action.add:
 87                details += f"{indent * 2}{f}\n"
 88            details += f"{indent}delete:\n"
 89            for f in action.delete:
 90                details += f"{indent * 2}{f}\n"
 91
 92        return details
 93
 94    def serialize(self, filepath: str = None):
 95        """Serializes the model into a json string.
 96
 97        Args:
 98            filepath (str):
 99                Optional; If supplied, the json string will be written to the
100                filepath.
101
102        Returns:
103            A string in json format representing the model.
104        """
105
106        serial = dumps(self._serialize(), cls=ComplexEncoder)
107        if filepath is not None:
108            with open(filepath, "w") as fp:
109                fp.write(serial)
110        return serial

112    def __to_tarski_formula(self, attribute: Set[str], lang: FirstOrderLanguage):
113        """Converts a set of strings (referencing an attribute of a LearnedAction, i.e. its preconditions)
114        to an Atom or CompoundFormula, in order to set up a tarski action.
115
116        Args:
117            attribute (Set[str]):
118                The attribute to be converted to an Atom or CompoundFormula.
119            lang (FirstOrderLanguage):
120                The relevant language.
121
122        Returns:
123            The attribute of the LearnedAction, converted to an Atom or CompoundFormula.
124        """
125        # return top if there are no constraints
126        if not attribute:
127            return top
128        # creates Atom
129        elif len(attribute) == 1:
130            return lang.get(list(attribute)[0].replace(" ", "_"))()  # type: ignore
131        # creates CompoundFormula
132        else:
133            return CompoundFormula(
134                Connective.And, [lang.get(a.replace(" ", "_"))() for a in attribute]  # type: ignore
135            )
